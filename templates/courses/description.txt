    r = Review(
        rating=i % 6,
        text=f"Отзыв {i}",
        user_id=users[i % len(users)].id,
        course_id=course.id
    )
    db.session.add(r)
    course.rating_sum += r.rating
    course.rating_num += 1
db.session.commit()
